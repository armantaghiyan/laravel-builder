<?php

namespace App\Http\Resources\Admin\{model};

use App\Core\Domain\Common\Constants\Rk;
use App\Http\Resources\ResponseManager;
use Illuminate\Http\Request;
use Illuminate\Http\Resources\Json\JsonResource;

class {model-action}Resource extends JsonResource {

    public function __construct(
   		public $items,
   		public $count,
   	) {
   		parent::__construct($items);
   	}

    /**
     * Transform the resource into an array.
     *
     * @return array<string, mixed>
     */
    public function toArray(Request $request): array {

        return (new ResponseManager())->cast([
            Rk::ITEMS => {model}Resource::collection($this->items),
            Rk::COUNT => $this->count,
        ]);
    }
}
